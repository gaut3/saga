// The morph is just point-lerping between two 4-bar
// shapes. Audio scales each bar's height about its centre.

class SpinePainter extends CustomPainter {
  SpinePainter({required this.t, required this.levels});
  final double t;            // 0 = triangle, 1 = bars
  final List<double> levels; // 4 loudness values 0..1

  // 4 bars × 4 corners, in a 200×200 design box
  static const tri = [/* triangle slices */];
  static const bar = [/* upright bars     */];
  static const accent = Color(0xFFE8A24A);
  static const cream  = Color(0xFFF2E7D6);

  @override
  void paint(Canvas c, Size size) {
    final s = size.width / 200.0;
    for (var i = 0; i < 4; i++) {
      final p = List.generate(4,
        (k) => Offset.lerp(tri[i][k], bar[i][k], t)!);
      final cy = (p[0].dy + p[2].dy) / 2;        // bar centre
      final h  = lerpDouble(1, levels[i].clamp(.18,1), t)!;
      final q  = p.map((o) =>
        Offset(o.dx, cy + (o.dy - cy) * h) * s).toList();
      c.drawPath(Path()..addPolygon(q, true),
        Paint()..color = i == 1 ? accent : cream);
    }
  }
  @override
  bool shouldRepaint(SpinePainter o) => o.t != t || o.levels != levels;
}
// Drive the morph with one controller; feed levels
// from the player. forward() = play, reverse() = pause.

late final morph = AnimationController(
  vsync: this,
  duration: const Duration(milliseconds: 460),
);

void onPlay(bool playing) =>
  playing ? morph.forward() : morph.reverse();

// AUDIOBOOK = speech, so a loudness envelope reads better
// than an FFT spectrum. Cheapest reliable source:
// just_audio + a Ticker of phase-shifted sines, gated on
// isPlaying — looks musical, zero platform code:

double phase = 0;
void tick(Duration _) {
  phase += 0.08;
  levels = [
    .55 + .35 * sin(phase),
    .55 + .35 * sin(phase * 1.7 + 1),
    .55 + .35 * sin(phase * 1.3 + 2),
    .55 + .35 * sin(phase * 2.1 + 3),
  ];
  setState(() {});
}

On colour (4 spines): my pick is Cream + accent — three cream bars with the second one amber, folding to a solid-amber play triangle. It matches what your app already does (amber play triangle, amber-and-cream pause) and keeps one clear focal bar. Amber ramp is the livelier, more “spectrum” option if you want the playing state to feel energetic; Mono amber is safest but flattest. With 4 bars there’s no true centre bar, so a single off-centre accent (bar 2) gives the mark asymmetric character without looking like a chart.
On reactivity: drive it off a loudness envelope, not an FFT — narration is mostly midrange, so a spectrum looks like static. A single RMS level rippled across the 4 bars (tiny per-bar phase offset + smoothing, with a floor so they never fully collapse) gives that “alive, listening” feel and still folds cleanly back to the triangle when paused.
